```typescript
// src/app/components/ide/theme/CSSVariableInjector.ts
export class CSSVariableInjector {
  private static instance: CSSVariableInjector;
  private currentVariables: Map<string, string> = new Map();
  private pendingUpdate: boolean = false;
  
  // 批量更新CSS变量（性能优化）
  batchUpdate(variables: Record<string, string>): void {
    // 只更新变化的变量
    const changes = this.detectChanges(variables);
    if (Object.keys(changes).length > 0) {
      this.applyChanges(changes);
    }
  }
  
  // 检测变化
  private detectChanges(newVariables: Record<string, string>): Record<string, string> {
    const changes: Record<string, string> = {};
    for (const [key, value] of Object.entries(newVariables)) {
      if (this.currentVariables.get(key) !== value) {
        changes[key] = value;
      }
    }
    return changes;
  }
  
  // 应用变化
  private applyChanges(changes: Record<string, string>): void {
    const root = document.documentElement;
    for (const [key, value] of Object.entries(changes)) {
      root.style.setProperty(key, value);
      this.currentVariables.set(key, value);
    }
  }
}
```

```typescript
    };
```

```typescript
// src/app/components/ide/theme/SystemThemeListener.ts
export class SystemThemeListener {
  private mediaQuery: MediaQueryList;
  private listeners: Set<SystemThemeCallback> = new Set();
  
  constructor() {
    this.mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
    this.mediaQuery.addEventListener('change', this.handleChange);
  }
  
  // 获取系统主题
  getSystemTheme(): 'dark' | 'light' {
    return this.mediaQuery.matches ? 'dark' : 'light';
  }
  
  // 监听系统主题变化
  private handleChange = (e: MediaQueryListEvent): void => {
    const theme = e.matches ? 'dark' : 'light';
    this.notifyListeners(theme);
  };
  
  // 清理资源
  destroy(): void {
    this.mediaQuery.removeEventListener('change', this.handleChange);
    this.listeners.clear();
  }
}
```

```typescript
// src/app/components/ide/preview/PreviewErrorHandler.ts
export class PreviewErrorHandler {
  private errors: ConsoleEntry[] = [];
  private maxEntries: number = 1000;
  
  // 捕获错误
  captureError(error: Error, type: ConsoleEntryType): void {
    const entry: ConsoleEntry = {
      id: generateId(),
      type,
      message: error.message,
      timestamp: Date.now(),
      stack: error.stack
    };
    
    this.errors.push(entry);
    this.trimEntries();
  }
  
  // 过滤日志
  filterLogs(level?: ConsoleEntryType, keyword?: string): ConsoleEntry[] {
    let filtered = this.errors;
    
    if (level) {
      filtered = filtered.filter(e => e.type === level);
    }
    
    if (keyword) {
      filtered = filtered.filter(e => 
        e.message.toLowerCase().includes(keyword.toLowerCase())
      );
    }
    
    return filtered;
  }
  
  // 清空日志
  clearLogs(): void {
    this.errors = [];
  }
  
  // 限制日志数量
  private trimEntries(): void {
    if (this.errors.length > this.maxEntries) {
      this.errors = this.errors.slice(-this.maxEntries);
    }
  }
}
```

```typescript
export class ConversationHistory {
  private messages: ConversationMessage[] = [];
  private maxMessages: number = 50;
  private maxTokens: number = 8000;
  
  // 添加消息
  addMessage(message: ConversationMessage): void {
    this.messages.push({
      ...message,
      timestamp: Date.now()
    });
    
    this.trimMessages();
  }
  
  // 获取消息（应用限制）
  getMessages(limit?: { maxMessages?: number; maxTokens?: number }): ConversationMessage[] {
    let messages = this.messages;
    
    // 应用消息数量限制
    const maxMsgs = limit?.maxMessages ?? this.maxMessages;
    if (messages.length > maxMsgs) {
      messages = messages.slice(-maxMsgs);
    }
    
    // 应用Token数量限制
    const maxToks = limit?.maxTokens ?? this.maxTokens;
    messages = this.limitByTokens(messages, maxToks);
    
    return messages;
  }
  
  // 搜索消息
  searchMessages(query: string): ConversationMessage[] {
    const lowerQuery = query.toLowerCase();
    
    return this.messages.filter(msg => 
      msg.content.toLowerCase().includes(lowerQuery) ||
      msg.role.toLowerCase().includes(lowerQuery)
    );
  }
  
  // 按Token限制消息
  private limitByTokens(messages: ConversationMessage[], maxTokens: number): ConversationMessage[] {
    let totalTokens = 0;
    const result: ConversationMessage[] = [];
    
    // 从后向前遍历
    for (let i = messages.length - 1; i >= 0; i--) {
      const msg = messages[i];
      const tokens = this.estimateTokens(msg.content);
      
      if (totalTokens + tokens > maxTokens) {
        break;
      }
      
      result.unshift(msg);
      totalTokens += tokens;
    }
    
    return result;
  }
  
  // 估算Token数量
  private estimateTokens(text: string): number {
    // 简单估算：平均4个字符 = 1个Token
    return Math.ceil(text.length / 4);
  }
}
```

```typescript
export class ContextCompressor {
  private static MIN_COMPRESS_SIZE = 1000; // 最小压缩大小
  private static TARGET_COMPRESSION_RATIO = 0.5; // 目标压缩率
  
  // 压缩上下文
  compress(content: string, maxTokens: number): CompressionResult {
    // 小内容不压缩
    if (content.length < this.MIN_COMPRESS_SIZE) {
      return {
        compressed: content,
        originalSize: content.length,
        compressedSize: content.length,
        ratio: 1.0,
        strategy: 'none'
      };
    }
    
    // 分析代码结构
    const structure = this.analyzeStructure(content);
    
    // 选择压缩策略
    const strategy = this.selectStrategy(structure, maxTokens);
    
    // 执行压缩
    const compressed = this.executeCompression(content, strategy);
    
    return {
      compressed,
      originalSize: content.length,
      compressedSize: compressed.length,
      ratio: compressed.length / content.length,
      strategy: strategy.name
    };
  }
  
  // 分析代码结构
  private analyzeStructure(content: string): CodeStructure {
    return {
      functions: this.extractFunctions(content),
      classes: this.extractClasses(content),
      imports: this.extractImports(content),
      comments: this.extractComments(content),
      testCode: this.extractTestCode(content)
    };
  }
  
  // 选择压缩策略
  private selectStrategy(structure: CodeStructure, maxTokens: number): CompressionStrategy {
    const totalTokens = this.estimateTokens(structure);
    
    if (totalTokens <= maxTokens * 0.8) {
      return { name: 'light', removeComments: false, keepTests: true };
    } else if (totalTokens <= maxTokens * 1.5) {
      return { name: 'medium', removeComments: true, keepTests: false };
    } else {
      return { name: 'aggressive', removeComments: true, keepTests: false, useSignatures: true };
    }
  }
  
  // 执行压缩
  private executeCompression(content: string, strategy: CompressionStrategy): string {
    let result = content;
    
    // 移除注释
    if (strategy.removeComments) {
      result = this.removeComments(result);
    }
    
    // 移除测试代码
    if (!strategy.keepTests) {
      result = this.removeTestCode(result);
    }
    
    // 使用签名
    if (strategy.useSignatures) {
      result = this.replaceWithSignatures(result);
    }
    
    return result;
```

```typescript
    const measure = performance.getEntriesByName(name)[0];
    return measure.duration;
  }
  
  // 生成性能报告
  generateReport(): PerformanceReport {
    const metrics = Array.from(this.metrics.values());
    
    return {
      timestamp: Date.now(),
      metrics,
      summary: {
        averageRenderTime: this.calculateAverage(metrics, 'render-time'),
        averageMemoryUsage: this.calculateAverage(metrics, 'memory-usage'),
        bottlenecks: this.identifyBottlenecks(metrics)
      },
      recommendations: this.generateRecommendations(metrics)
    };
  }
  
  // 识别性能瓶颈
  private identifyBottlenecks(metrics: PerformanceMetric[]): PerformanceBottleneck[] {
    const bottlenecks: PerformanceBottleneck[] = [];
    
    metrics.forEach(metric => {
      if (metric.name === 'render-time' && metric.value > 100) {
        bottlenecks.push({
          type: 'render',
          severity: 'high',
          message: `Render time is too slow: ${metric.value}ms`,
          suggestion: 'Consider using virtual scrolling or reducing component complexity'
        });
      }
    });
    
    return bottlenecks;
  }
}
```

```typescript
// src/app/components/ide/plugins/PluginManager.ts
export interface IPlugin {
  id: string;
  name: string;
  version: string;
  dependencies?: string[];
  
  // 生命周期
  activate(context: PluginContext): Promise<void>;
  deactivate(): Promise<void>;
}

export interface PluginContext {
  // 核心API
  ide: IDEAPI;
  
  // UI扩展
  ui: UIAPI;
  
  // 数据访问
  data: DataAPI;
  
  // 事件系统
  events: EventAPI;
}

export class PluginManager {
  private plugins: Map<string, IPlugin> = new Map();
  private contexts: Map<string, PluginContext> = new Map();
  
  // 加载插件
  async loadPlugin(plugin: IPlugin): Promise<void> {
    // 验证插件
    this.validatePlugin(plugin);
    
    // 检查依赖
    await this.checkDependencies(plugin);
    
    // 创建上下文
    const context = this.createContext(plugin);
    
    // 激活插件
    await plugin.activate(context);
    
    // 注册插件
    this.plugins.set(plugin.id, plugin);
    this.contexts.set(plugin.id, context);
  }
  
  // 卸载插件
  async unloadPlugin(pluginId: string): Promise<void> {
    const plugin = this.plugins.get(pluginId);
    if (!plugin) return;
    
    // 停用插件
    await plugin.deactivate();
    
    // 清理资源
    this.plugins.delete(pluginId);
    this.contexts.delete(pluginId);
  }
  
  // 创建插件上下文
  private createContext(plugin: IPlugin): PluginContext {
    return {
      ide: this.createIDEAPI(plugin),
      ui: this.createUIAPI(plugin),
      data: this.createDataAPI(plugin),
      events: this.createEventAPI(plugin)
    };
  }
}
```

   - **缓解措施**: 自动化测试，CI集成
   - **应急预案**: 优先保证核心功能测试
