## 🚀 使用示例

### 基础使用

```typescript
import { CodeValidator } from './CodeValidator';

const validator = new CodeValidator();

const block = {
  filepath: 'test.ts',
  content: 'const x = 1;',
  language: 'ts',
  isNewFile: true
};

const result = validator.validate(block);

console.log(result.valid);      // true
console.log(result.warnings);   // []
console.log(result.errors);     // []
console.log(result.metrics);    // { lines: 1, characters: 12, complexity: 'low' }
```

### 批量验证

```typescript
const blocks = [
  { filepath: 'file1.ts', content: 'const x = 1;', language: 'ts', isNewFile: true },
  { filepath: 'file2.ts', content: 'const y = 2;', language: 'ts', isNewFile: true }
];

const results = validator.validateAll(blocks);
// Map { 'file1.ts' => {...}, 'file2.ts' => {...} }
```
