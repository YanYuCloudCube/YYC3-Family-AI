## 📝 使用示例

### 基本使用
```typescript
import { recognizeIntent } from './llm/IntentRecognitionEngine';

// 识别意图
const result = recognizeIntent('帮我创建一个TypeScript组件');

console.log(result.primaryIntent.type);  // 'create'
console.log(result.primaryIntent.confidence);  // 0.85
console.log(result.primaryIntent.features.entities);  // [{type: 'language', value: 'TypeScript'}]
```

### 高级使用
```typescript
import { IntentRecognitionEngine } from './llm/IntentRecognitionEngine';

// 创建引擎实例
const engine = new IntentRecognitionEngine({
  enableMultiIntent: true,
  confidenceThreshold: 0.5,
  maxCandidates: 5,
});

// 识别意图
const result = engine.recognize('创建组件并优化性能', {
  currentFile: '/src/app.ts',
  projectContext: 'React项目',
});

// 获取详细报告
const report = engine.generateReport(result);
console.log(report);

// 获取统计信息
const stats = engine.getStatistics();
console.log(stats);
```

### 自定义模式
```typescript
import { IntentRecognitionEngine, IntentType, IntentPriority, IntentCategory } from './llm';

const engine = new IntentRecognitionEngine();

// 添加自定义意图模式
engine.addCustomPattern({
  type: IntentType.CREATE,
  patterns: [/快速创建|一键生成/i],
  keywords: ['快速创建', '一键生成'],
  priority: IntentPriority.HIGH,
  category: IntentCategory.CODE_OPERATION,
  description: '快速创建模式',
});

// 使用自定义模式
const result = engine.recognize('快速创建一个组件');
console.log(result.primaryIntent.type);  // 'create'
```
