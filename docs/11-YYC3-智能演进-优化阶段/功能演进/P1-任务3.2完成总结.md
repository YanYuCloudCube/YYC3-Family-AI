## 🔧 使用示例

```typescript
import { TaskExtractionEngine } from './llm/TaskExtractionEngine';

// 创建引擎
const engine = new TaskExtractionEngine({
  maxTasks: 20,
  minConfidence: 0.5,
  enableDeduplication: true,
});

// 从文本提取任务
const result = engine.extract(`
  TODO: 创建用户组件 P1
  - [ ] 实现登录功能
  - [x] 完成注册流程
  1. 编写测试用例
`);

console.log(result.tasks);
// [
//   { title: '创建用户组件', priority: 'high', type: 'feature', ... },
//   { title: '实现登录功能', priority: 'medium', type: 'feature', ... },
//   ...
// ]

// 从AI响应提取
const aiResult = engine.extractFromAIResponse(

);
```
