```typescript
// 导入 CodeValidator
import { CodeValidator, type ValidationResult } from "../CodeValidator";

// 扩展接口
export interface CodeValidationResult {
  filepath: string;
  result: ValidationResult;
}

export interface ParseAndValidateResult {
  plan: CodeApplicationPlan;
  validations: Map<string, ValidationResult>;
  hasErrors: boolean;
  hasWarnings: boolean;
}

// 实现验证功能
export function validateCodeBlock(block: ParsedCodeBlock): ValidationResult {
  const validator = new CodeValidator();
  return validator.validate(block);
}

export function validateCodeBlocks(
  blocks: ParsedCodeBlock[]
): Map<string, ValidationResult> {
  const validator = new CodeValidator();
  return validator.validateAll(blocks);
}

export function parseAndValidateCodeBlocks(
  llmResponse: string,
  existingFiles: Record<string, string>
): ParseAndValidateResult {
  const plan = parseCodeBlocks(llmResponse, existingFiles);
  const validations = validateCodeBlocks(plan.blocks);
  
  let hasErrors = false;
  let hasWarnings = false;
  for (const result of validations.values()) {
    if (result.errors.length > 0) hasErrors = true;
    if (result.warnings.length > 0) hasWarnings = true;
  }
  
  return { plan, validations, hasErrors, hasWarnings };
}
```

```typescript
    validationResult?: ParseAndValidateResult
  ) => void;
  onError: (error: string) => void;
  onContextReady?: (ctx: ProjectContext, intent: UserIntent) => void;
}

// 修改 runPipeline 函数
export function runPipeline(...) {
  // ...
  chatCompletionStream(
    input.provider,
    input.modelId,
    llmMessages,
    {
      onToken: callbacks.onToken,
      onDone: (fullText) => {
        // Parse & validate code blocks
        const result = parseAndValidateCodeBlocks(fullText, input.fileContents);
        
        if (result.plan.blocks.length > 0) {
          codePlan = result.plan;
          validationResult = result;
          
          // Log validation results
          if (result.hasErrors) {
            console.error("[AIPipeline] Code validation found errors");
          }
          if (result.hasWarnings) {
            console.warn("[AIPipeline] Code validation found warnings");
          }
        }
        
        callbacks.onDone(fullText, codePlan, validationResult);
      },
      // ...
    }
  );
}
```
