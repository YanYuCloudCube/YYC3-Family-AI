## 一、实时预览模式控制系统

### 1.1 功能需求分析

根据测试用例，需要实现三种预览模式：
- **实时模式**: 文件修改立即触发预览更新
- **手动模式**: 需要手动触发预览更新
- **延迟模式**: 文件修改后延迟一定时间再更新

### 1.2 设计方案

#### 1.2.1 预览模式控制器类

```typescript
/**
 * @file PreviewModeController.ts
 * @description 预览模式控制器，管理实时/手动/延迟三种预览模式
 */

export class PreviewModeController {
  private mode: PreviewMode = "realtime";
  private delayTimer: ReturnType<typeof setTimeout> | null = null;
  private pendingUpdate: boolean = false;
  private readonly DEFAULT_DELAY = 500; // 默认延迟500ms

  constructor(
    private onTriggerUpdate: () => void,
    private delay: number = this.DEFAULT_DELAY
  ) {}

  /**
   * 设置预览模式
   */
  setMode(mode: PreviewMode): void {
    this.mode = mode;
    this.clearPendingUpdate();
  }

  /**
   * 文件变更处理
   */
  handleFileChange(): void {
    switch (this.mode) {
      case "realtime":
        this.triggerImmediateUpdate();
        break;
      case "manual":
        this.pendingUpdate = true;
        break;
      case "delayed":
        this.scheduleDelayedUpdate();
        break;
    }
  }

  /**
   * 手动触发更新（手动模式）
   */
  manualTrigger(): void {
    if (this.mode === "manual" || this.pendingUpdate) {
      this.triggerImmediateUpdate();
      this.pendingUpdate = false;
    }
  }

  /**
   * 立即触发更新
   */
  private triggerImmediateUpdate(): void {
    this.clearPendingUpdate();
    this.onTriggerUpdate();
  }

  /**
   * 调度延迟更新
   */
  private scheduleDelayedUpdate(): void {
    this.clearPendingUpdate();
    this.delayTimer = setTimeout(() => {
      this.triggerImmediateUpdate();
    }, this.delay);
  }

  /**
   * 清除待处理的更新
   */
  private clearPendingUpdate(): void {
    if (this.delayTimer) {
      clearTimeout(this.delayTimer);
      this.delayTimer = null;
    }
  }

  /**
   * 设置延迟时间
   */
  setDelay(delay: number): void {
    this.delay = Math.max(100, Math.min(5000, delay)); // 限制100ms-5s
  }

  /**
   * 销毁控制器
   */
  destroy(): void {
    this.clearPendingUpdate();
  }
}
```

#### 2.2.1 快照管理器类

```typescript
/**
 * @file SnapshotManager.ts
 * @description 快照管理器，负责创建、存储、恢复、比较预览快照
 */

export interface Snapshot {
  id: string;
  label: string;
  timestamp: number;
  files: Array<{
    path: string;
    content: string;
    hash: string; // 内容哈希用于快速比较
  }>;
  metadata: {
    totalFiles: number;
    totalLines: number;
    device?: DevicePreset;
    zoom?: number;
  };
}

export class SnapshotManager {
  private snapshots: Map<string, Snapshot> = new Map();
  private readonly MAX_SNAPSHOTS = 50;

  constructor() {
    this.loadFromStorage();
  }

  /**
   * 创建快照
   */
  createSnapshot(
    label: string,
    files: Array<{ path: string; content: string }>,
    metadata?: Partial<Snapshot["metadata"]>
  ): Snapshot {
    const id = `snap_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`;
    
    const snapshot: Snapshot = {
      id,
      label,
      timestamp: Date.now(),
      files: files.map(f => ({
        ...f,
        hash: this.calculateHash(f.content)
      })),
      metadata: {
        totalFiles: files.length,
        totalLines: files.reduce((sum, f) => 
          sum + f.content.split('\n').length, 0),
        ...metadata
      }
    };

    this.snapshots.set(id, snapshot);
    this.enforceLimit();
    this.saveToStorage();

    return snapshot;
  }

  /**
   * 列出所有快照
   */
  listSnapshots(): Snapshot[] {
    return Array.from(this.snapshots.values())
      .sort((a, b) => b.timestamp - a.timestamp);
  }

  /**
   * 获取单个快照
   */
  getSnapshot(id: string): Snapshot | null {
    return this.snapshots.get(id) || null;
  }

  /**
   * 恢复快照
   */
  restoreSnapshot(
    id: string,
    applyFn: (files: Array<{ path: string; content: string }>) => void
  ): boolean {
    const snapshot = this.snapshots.get(id);
    if (!snapshot) return false;

    applyFn(snapshot.files);
    return true;
  }

  /**
   * 删除快照
   */
  deleteSnapshot(id: string): boolean {
    const deleted = this.snapshots.delete(id);
    if (deleted) {
      this.saveToStorage();
    }
    return deleted;
  }

  /**
   * 比较两个快照
   */
  compareSnapshots(id1: string, id2: string): {
    added: string[];
    removed: string[];
    modified: string[];
    unchanged: string[];
  } {
    const snap1 = this.snapshots.get(id1);
    const snap2 = this.snapshots.get(id2);

    if (!snap1 || !snap2) {
      throw new Error("Snapshot not found");
    }

    const files1 = new Map(snap1.files.map(f => [f.path, f]));
    const files2 = new Map(snap2.files.map(f => [f.path, f]));

    const added: string[] = [];
    const removed: string[] = [];
    const modified: string[] = [];
    const unchanged: string[] = [];

    // 检查新增和修改
    for (const [path, file2] of files2) {
      const file1 = files1.get(path);
      if (!file1) {
        added.push(path);
      } else if (file1.hash !== file2.hash) {
        modified.push(path);
      } else {
        unchanged.push(path);
      }
    }

    // 检查删除
    for (const path of files1.keys()) {
      if (!files2.has(path)) {
        removed.push(path);
      }
    }

    return { added, removed, modified, unchanged };
  }

  /**
   * 计算内容哈希（简单实现）
   */
  private calculateHash(content: string): string {
    let hash = 0;
    for (let i = 0; i < content.length; i++) {
      const char = content.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash; // Convert to 32bit integer
    }
    return hash.toString(36);
  }

  /**
   * 强制限制快照数量
   */
  private enforceLimit(): void {
    if (this.snapshots.size > this.MAX_SNAPSHOTS) {
      const sorted = this.listSnapshots();
      const toRemove = sorted.slice(this.MAX_SNAPSHOTS);
      for (const snap of toRemove) {
        this.snapshots.delete(snap.id);
      }
    }
  }

  /**
   * 持久化到 localStorage
   */
  private saveToStorage(): void {
    try {
      const data = Array.from(this.snapshots.values());
      localStorage.setItem('yyc3_snapshots', JSON.stringify(data));
    } catch (error) {
      console.error('[SnapshotManager] Failed to save:', error);
    }
  }

  /**
   * 从 localStorage 加载
   */
  private loadFromStorage(): void {
    try {
      const data = localStorage.getItem('yyc3_snapshots');
      if (data) {
        const snapshots = JSON.parse(data) as Snapshot[];
        for (const snap of snapshots) {
          this.snapshots.set(snap.id, snap);
        }
      }
    } catch (error) {
      console.error('[SnapshotManager] Failed to load:', error);
    }
  }
}
```

#### 3.2.1 代码验证器类

```typescript
/**
 * @file CodeValidator.ts
 * @description 代码验证器，检查生成代码的质量和安全性
 */

export interface ValidationResult {
  valid: boolean;
  warnings: string[];
  errors: string[];
  suggestions: string[];
  metrics: {
    lines: number;
    characters: number;
    complexity: "low" | "medium" | "high";
  };
}

export class CodeValidator {
  private readonly MAX_FILE_LENGTH = 10000; // 最大文件字符数
  private readonly MAX_LINE_COUNT = 500; // 最大行数

  /**
   * 验证代码块
   */
  validate(block: ParsedCodeBlock): ValidationResult {
    const warnings: string[] = [];
    const errors: string[] = [];
    const suggestions: string[] = [];

    // 1. 空代码检测
    if (!block.content || block.content.trim().length === 0) {
      errors.push("代码为空");
      return this.buildResult(false, warnings, errors, suggestions, block.content);
    }

    // 2. 长度检查
    const lengthCheck = this.checkLength(block.content);
    warnings.push(...lengthCheck.warnings);
    errors.push(...lengthCheck.errors);

    // 3. 语法验证
    const syntaxCheck = this.checkSyntax(block.content, block.language);
    errors.push(...syntaxCheck.errors);
    warnings.push(...syntaxCheck.warnings);

    // 4. 安全性检查
    const securityCheck = this.checkSecurity(block.content);
    warnings.push(...securityCheck.warnings);
    errors.push(...securityCheck.errors);

    // 5. 最佳实践检查
    const practiceCheck = this.checkBestPractices(block.content, block.language);
    suggestions.push(...practiceCheck.suggestions);
    warnings.push(...practiceCheck.warnings);

    // 6. 计算指标
    const metrics = this.calculateMetrics(block.content);

    return this.buildResult(
      errors.length === 0,
      warnings,
      errors,
      suggestions,
      block.content,
      metrics
    );
  }

  /**
   * 长度检查
   */
  private checkLength(content: string): { warnings: string[]; errors: string[] } {
    const warnings: string[] = [];
    const errors: string[] = [];

    if (content.length > this.MAX_FILE_LENGTH) {
      errors.push(`代码过长 (${content.length} 字符，建议不超过 ${this.MAX_FILE_LENGTH})`);
    } else if (content.length > this.MAX_FILE_LENGTH * 0.8) {
      warnings.push(`代码接近长度限制 (${content.length}/${this.MAX_FILE_LENGTH})`);
    }

    const lineCount = content.split('\n').length;
    if (lineCount > this.MAX_LINE_COUNT) {
      warnings.push(`文件行数较多 (${lineCount} 行)，建议拆分模块`);
    }

    return { warnings, errors };
  }

  /**
   * 语法验证（基础检查）
   */
  private checkSyntax(
    content: string,
    language: string
  ): { errors: string[]; warnings: string[] } {
    const errors: string[] = [];
    const warnings: string[] = [];

    // 检查括号匹配
    const bracketCheck = this.checkBrackets(content);
    if (!bracketCheck.valid) {
      errors.push(`括号不匹配: ${bracketCheck.message}`);
    }

    // TypeScript/JavaScript 特定检查
    if (['ts', 'tsx', 'js', 'jsx'].includes(language)) {
      // 检查未闭合的字符串
      if (this.hasUnclosedString(content)) {
        errors.push("存在未闭合的字符串");
      }

      // 检查是否包含 debugger
      if (content.includes('debugger')) {
        warnings.push("代码中包含 debugger 语句");
      }
    }

    return { errors, warnings };
  }

  /**
   * 安全性检查
   */
  private checkSecurity(content: string): { warnings: string[]; errors: string[] } {
    const warnings: string[] = [];
    const errors: string[] = [];

    // 检查危险模式
    const dangerousPatterns = [
      { pattern: /eval\s*\(/, message: "使用了 eval()，存在安全风险" },
      { pattern: /Function\s*\(/, message: "动态创建函数，可能存在风险" },
      { pattern: /innerHTML\s*=/, message: "使用 innerHTML，注意 XSS 风险" },
      { pattern: /dangerouslySetInnerHTML/, message: "使用 dangerouslySetInnerHTML，确保内容安全" },
      { pattern: /document\.write/, message: "使用 document.write，已废弃" },
    ];

    for (const { pattern, message } of dangerousPatterns) {
      if (pattern.test(content)) {
        warnings.push(message);
      }
    }

    // 检查硬编码的敏感信息
    const sensitivePatterns = [
      { pattern: /password\s*[:=]\s*["'].*["']/i, message: "可能包含硬编码密码" },
      { pattern: /api[_-]?key\s*[:=]\s*["'].*["']/i, message: "可能包含硬编码 API key" },
      { pattern: /secret\s*[:=]\s*["'].*["']/i, message: "可能包含硬编码密钥" },
    ];

    for (const { pattern, message } of sensitivePatterns) {
      if (pattern.test(content)) {
        errors.push(message);
      }
    }

    return { warnings, errors };
  }

  /**
   * 最佳实践检查
   */
  private checkBestPractices(
    content: string,
    language: string
  ): { warnings: string[]; suggestions: string[] } {
    const warnings: string[] = [];
    const suggestions: string[] = [];

    // 检查 console.log
    const consoleLogMatches = content.match(/console\.log/g);
    if (consoleLogMatches && consoleLogMatches.length > 5) {
      warnings.push(`存在多个 console.log (${consoleLogMatches.length} 个)，建议清理`);
    }

    // 检查 TODO/FIXME
    const todoMatches = content.match(/TODO|FIXME/gi);
    if (todoMatches && todoMatches.length > 0) {
      suggestions.push(`包含 ${todoMatches.length} 个待办事项 (TODO/FIXME)`);
    }

    // TypeScript 特定建议
    if (language === 'ts' || language === 'tsx') {
      if (content.includes(': any')) {
        suggestions.push("使用了 any 类型，建议使用具体类型");
      }
      if (!content.includes('export')) {
        suggestions.push("文件没有导出任何内容，确认是否为模块");
      }
    }

    // React 特定检查
    if (language === 'tsx' || language === 'jsx') {
      if (!content.includes('import React') && content.includes('useState')) {
        warnings.push("使用了 React hooks 但未导入 React");
      }
    }

    return { warnings, suggestions };
  }

  /**
   * 括号匹配检查
   */
  private checkBrackets(content: string): { valid: boolean; message: string } {
    const stack: string[] = [];
    const pairs: Record<string, string> = {
      '(': ')',
      '[': ']',
      '{': '}'
    };

    for (let i = 0; i < content.length; i++) {
      const char = content[i];
      if (pairs[char]) {
        stack.push(char);
      } else if (Object.values(pairs).includes(char)) {
        const last = stack.pop();
        if (!last || pairs[last] !== char) {
          return { valid: false, message: `位置 ${i} 处括号不匹配` };
        }
      }
    }

    if (stack.length > 0) {
      return { valid: false, message: `存在 ${stack.length} 个未闭合的括号` };
    }

    return { valid: true, message: "" };
  }

  /**
   * 检查未闭合的字符串
   */
  private hasUnclosedString(content: string): boolean {
    let inString = false;
    let stringChar = '';

    for (let i = 0; i < content.length; i++) {
      const char = content[i];
      const prevChar = i > 0 ? content[i - 1] : '';

      if (!inString && (char === '"' || char === "'" || char === '`')) {
        inString = true;
        stringChar = char;
      } else if (inString && char === stringChar && prevChar !== '\\') {
        inString = false;
      }
    }

    return inString;
  }

  /**
   * 计算代码指标
   */
  private calculateMetrics(content: string): ValidationResult["metrics"] {
    const lines = content.split('\n').length;
    const characters = content.length;

    // 简单的复杂度评估
    const complexityIndicators = [
      /if\s*\(/g,
      /for\s*\(/g,
      /while\s*\(/g,
      /switch\s*\(/g,
      /\?\s*:/g, // 三元运算符
    ];

    let complexityScore = 0;
    for (const pattern of complexityIndicators) {
      const matches = content.match(pattern);
      complexityScore += matches ? matches.length : 0;
    }

    const complexity = complexityScore < 5 ? "low" : 
                       complexityScore < 15 ? "medium" : "high";

    return { lines, characters, complexity };
  }

  /**
   * 构建结果
   */
  private buildResult(
    valid: boolean,
    warnings: string[],
    errors: string[],
    suggestions: string[],
    content: string,
    metrics?: ValidationResult["metrics"]
  ): ValidationResult {
    return {
      valid,
      warnings,
      errors,
      suggestions,
      metrics: metrics || {
        lines: content.split('\n').length,
        characters: content.length,
        complexity: "low"
      }
    };
  }
}
```

```typescript
/**
```

```typescript
 * @description 构建上下文感知的系统提示词，支持意图识别、上下文注入、Token控制

import { ProjectContext } from "./ContextCollector";

export type UserIntent =

export interface SystemPromptConfig {

  /**
   * 构建上下文部分
   */
  private static buildContextSection(
    context: ProjectContext,
    maxTokens: number
  ): string {
    const sections: string[] = [];
    let estimatedTokens = 0;

    // 文件树（始终包含）
    const fileTreeSection = `## 项目文件结构
\`\`\`
${context.fileTree}
\`\`\``;
    sections.push(fileTreeSection);
    estimatedTokens += this.estimateTokens(fileTreeSection);

    // Git 状态
    if (context.gitSummary) {
      const gitSection = `## Git 状态
- 当前分支: \`${context.gitSummary.branch}\`
- 已修改文件: ${context.gitSummary.changedFiles} 个
- 已暂存文件: ${context.gitSummary.stagedFiles} 个`;
      sections.push(gitSection);
      estimatedTokens += this.estimateTokens(gitSection);
    }

    // 活跃文件（最重要的上下文）
    if (context.activeFile && estimatedTokens < maxTokens * 0.6) {
      const activeFileSection = `## 当前编辑文件
**文件路径**: \`${context.activeFile.path}\`

\`\`\`${this.detectLanguage(context.activeFile.path)}
${context.activeFile.content}
\`\`\``;
      sections.push(activeFileSection);
      estimatedTokens += this.estimateTokens(activeFileSection);
    }

    // 打开的标签页
    if (context.openTabs.length > 0 && estimatedTokens < maxTokens * 0.8) {
      const tabsSection = `## 打开的文件
${context.openTabs.map(p => `- \`${p}\``).join('\n')}`;
      sections.push(tabsSection);
      estimatedTokens += this.estimateTokens(tabsSection);
    }

    // 其他文件内容（按需加载）
    if (context.selectedFilesContent && estimatedTokens < maxTokens * 0.9) {
      const remainingTokens = maxTokens - estimatedTokens;
      const otherFilesSection = this.buildOtherFilesSection(
        context.selectedFilesContent,
        remainingTokens
      );
      if (otherFilesSection) {
        sections.push(otherFilesSection);
      }
    }

    return sections.join('\n\n');
  }

  /**
   * 构建其他文件部分
   */
  private static buildOtherFilesSection(
    files: Record<string, string>,
    maxTokens: number
  ): string | null {
    const entries = Object.entries(files);
    if (entries.length === 0) return null;

    const sections: string[] = [];
    let estimatedTokens = 0;

    for (const [path, content] of entries) {
      const section = `### ${path}
\`\`\`${this.detectLanguage(path)}
${content}
\`\`\``;

      const tokens = this.estimateTokens(section);
      if (estimatedTokens + tokens > maxTokens) break;

      sections.push(section);
      estimatedTokens += tokens;
    }

    return sections.length > 0 
      ? `## 相关文件内容\n\n${sections.join('\n\n')}` 
      : null;
  }

  /**
   * 构建自定义指令
   */
  private static buildCustomInstructions(instructions: string): string {
    return `## 用户自定义指令
${instructions}`;
  }

  /**
   * 构建输出格式要求
   */
  private static buildOutputFormat(): string {
    return `## 代码输出格式要求

### 单文件输出
使用标准代码块，首行注明文件路径：
\`\`\`tsx
// filepath: src/components/MyComponent.tsx
import React from 'react'

export default function MyComponent() {
  // 组件实现
}
\`\`\`

### 多文件输出
每个文件使用独立代码块：
\`\`\`tsx
// filepath: src/components/Header.tsx
// ...
\`\`\`

\`\`\`tsx
// filepath: src/components/Footer.tsx
// ...
\`\`\`

### 重要规则
1. 始终提供完整的文件内容，不要使用 "...省略..." 
2. 文件路径必须清晰明确
3. 优先使用 TypeScript 类型定义
4. 遵循项目现有代码风格`;
  }

  /**
   * 检测语言类型
   */
  private static detectLanguage(filepath: string): string {
    const ext = filepath.split('.').pop()?.toLowerCase();
    const langMap: Record<string, string> = {
      'ts': 'typescript',
      'tsx': 'tsx',
      'js': 'javascript',
      'jsx': 'jsx',
      'css': 'css',
      'scss': 'scss',
      'json': 'json',
      'md': 'markdown',
      'html': 'html',
    };
    return langMap[ext || ''] || 'plaintext';
  }

  /**
   * 估算 Token 数（粗略）
   */
  private static estimateTokens(text: string): number {
    // 粗略估算: 英文 ~4 chars/token, 中文 ~2 chars/token
    // 这里使用简化算法
    const chineseChars = (text.match(/[\u4e00-\u9fa5]/g) || []).length;
    const otherChars = text.length - chineseChars;
    return Math.ceil(chineseChars / 2 + otherChars / 4);
  }
}

/**
 * 便捷函数：构建系统提示词
 */
export function buildSystemPrompt(
  intent: UserIntent,
  context: ProjectContext,
  customInstructions?: string
): string {
  return SystemPromptBuilder.build({
    intent,
    context,
    customInstructions,
    maxTokens: 8000
  });
}
```

```typescript
// 在 AIPipeline.ts 中集成所有功能

import { PreviewModeController } from "./PreviewModeController";

export function runPipeline(

  // 1. 收集上下文
  const context = collectContext({
    fileContents: input.fileContents,
    activeFile: input.activeFile,

    gitBranch: input.gitBranch,
    gitChanges: input.gitChanges
  });

  // 2. 检测意图
  const intent = detectIntent(input.userMessage);

  // 3. 构建系统提示词
  const systemPrompt = SystemPromptBuilder.build({
    intent,
    context,
    customInstructions: input.customInstructions,
    maxTokens: 8000
  });

  // 4. 调用 LLM
  const messages = [
    { role: "system", content: systemPrompt },
    ...input.conversationHistory,
    { role: "user", content: input.userMessage }
  ];

  chatCompletionStream(
    input.provider,
    input.modelId,
    messages,
    {
      onToken: callbacks.onToken,
      onDone: (fullText) => {
        // 5. 解析代码块
        const plan = parseCodeBlocks(fullText, input.fileContents);

        // 6. 验证代码
        const validator = new CodeValidator();
        for (const block of plan.blocks) {
          const result = validator.validate(block);
          if (!result.valid) {
            console.warn(`[CodeValidator] ${block.filepath}:`, result.errors);
          }
          if (result.warnings.length > 0) {
            console.warn(`[CodeValidator] ${block.filepath}:`, result.warnings);
          }
        }

        // 7. 应用代码
        if (callbacks.onDone) {
          callbacks.onDone(fullText, plan);
        }
      },
      onError: callbacks.onError
    }
  );
}
```
