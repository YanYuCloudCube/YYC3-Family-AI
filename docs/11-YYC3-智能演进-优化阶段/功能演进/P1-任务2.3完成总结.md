## 📝 使用示例

### 错误捕获使用
```typescript
import { getPreviewErrorCapturer, ErrorType, ErrorLevel } from './preview/PreviewErrorCapturer';

// 获取实例
const capturer = getPreviewErrorCapturer({
  maxEntries: 500,
  ignorePatterns: [/Extension context invalidated/]
});

// 启动捕获
capturer.start();

// 添加监听器
capturer.addListener((error) => {
  console.log('捕获到错误:', error.message);
});

// 手动添加错误
capturer.captureManualError(
  ErrorType.NETWORK,
  'Network request failed',
  ErrorLevel.ERROR,
  { url: 'https://api.example.com' }
);

// 获取统计
const stats = capturer.getStats();
console.log('错误统计:', stats);

// 导出错误
const exported = capturer.exportErrors();
```

### 控制台管理使用
```typescript
import { getConsoleManager, LogType } from './preview/ConsoleManager';

// 获取实例
const manager = getConsoleManager({ maxEntries: 500 });

// 添加日志
manager.addLog(LogType.ERROR, 'Error occurred');
manager.addLog(LogType.WARN, 'Warning message');

// 过滤日志
const errors = manager.filterByLevel(ErrorLevel.ERROR);

// 搜索日志
const results = manager.search('error');

// 导出日志
const html = manager.exportHTML();
const json = manager.exportJSON();
const text = manager.exportText();
```

### 错误过滤使用
```typescript
import { ErrorFilter } from './preview/ErrorFilter';

// 过滤错误
const filtered = ErrorFilter.filter(errors, {
  levels: [ErrorLevel.ERROR],
  sources: [ErrorSource.USER_CODE],
  keyword: 'timeout',
  startTime: Date.now() - 3600000
});

// 分组
const groups = ErrorFilter.group(errors, {
  groupBy: 'type',
  sortBy: 'count',
  sortOrder: 'desc'
});

// 统计
const stats = ErrorFilter.getStatistics(errors);

// 导出
const csv = ErrorFilter.exportFiltered(errors, 'csv');
const markdown = ErrorFilter.exportFiltered(errors, 'markdown');
```
