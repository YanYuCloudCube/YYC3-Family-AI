    - name: Checkout repository

        queries: security-extended,security-and-quality

      with:
        category: "/language:${{matrix.language}}"

/**
 * @file CryptoService.test.ts
 * @description 加密解密服务单元测试
 * @author YanYuCloudCube Team
 * @version v1.0.0
 * @created 2026-04-16
 */

import { describe, it, expect, vi, beforeEach } from 'vitest';
import { cryptoService } from '../CryptoService';

describe('CryptoService', () => {
  beforeEach(() => {
    cryptoService.reset();
  });

  describe('对称加密 (AES-GCM)', () => {
    const plaintext = 'Sensitive data to encrypt';
    
    it('应正确加密数据', async () => {
      const key = await cryptoService.generateKey();
      const encrypted = await cryptoService.encrypt(plaintext, key);
      
      expect(encrypted).toBeDefined();
      expect(encrypted.iv).toBeDefined();
      expect(encrypted.data).not.toBe(plaintext);
      expect(encrypted.data.length).toBeGreaterThan(0);
    });

    it('应正确解密数据', async () => {
      const key = await cryptoService.generateKey();
      const encrypted = await cryptoService.encrypt(plaintext, key);
      const decrypted = await cryptoService.decrypt(encrypted, key);
      
      expect(decrypted).toBe(plaintext);
    });

    it('不同IV应产生不同密文', async () => {
      const key = await cryptoService.generateKey();
      
      const encrypted1 = await cryptoService.encrypt(plaintext, key);
      const encrypted2 = await cryptoService.encrypt(plaintext, key);
      
      expect(encrypted1.data).not.toBe(encrypted2.data);
      expect(encrypted1.iv).not.toBe(encrypted2.iv);
    });

    it('应处理空字符串输入', async () => {
      const key = await cryptoService.generateKey();
      const encrypted = await cryptoService.encrypt('', key);
      const decrypted = await cryptoService.decrypt(encrypted, key);
      
      expect(decrypted).toBe('');
    });

    it('应处理Unicode字符', async () => {
      const unicodeText = '中文测试 🎉 Emoji 特殊字符 <script>alert("xss")</script>';
      const key = await cryptoService.generateKey();
      
      const encrypted = await cryptoService.encrypt(unicodeText, key);
      const decrypted = await cryptoService.decrypt(encrypted, key);
      
      expect(decrypted).toBe(unicodeText);
    });
  });

  describe('密钥管理', () => {
    it('应生成安全的随机密钥', async () => {
      const key1 = await cryptoService.generateKey();
      const key2 = await cryptoService.generateKey();
      
      // 密钥应该不同
      expect(key1).not.toBe(key2);
      
      // 密钥长度应该是256位 (32字节)
      const keyData = await cryptoService.exportKey(key1);
      expect(keyData.length).toBe(32); // 256 bits
    });

    it('应正确导出和导入密钥', async () => {
      const originalKey = await cryptoService.generateKey();
      const exported = await cryptoService.exportKey(originalKey);
      const importedKey = await cryptoService.importKey(exported);
      
      // 使用导入的密钥加密解密验证
      const plaintext = 'Test import/export';
      const encrypted = await cryptoService.encrypt(plaintext, importedKey);
      const decrypted = await cryptoService.decrypt(encrypted, importedKey);
      
      expect(decrypted).toBe(plaintext);
    });

    it('应安全存储密钥（使用包装密钥）', async () => {
      const masterKey = await cryptoService.generateMasterKey();
      const dataKey = await cryptoService.generateKey();
      
      const wrappedKey = await cryptoService.wrapKey(dataKey, masterKey);
      const unwrappedKey = await cryptoService.unwrapKey(wrappedKey, masterKey);
      
      // 验证解包后的密钥可用
      const encrypted = await cryptoService.encrypt('test', unwrappedKey);
      const decrypted = await cryptoService.decrypt(encrypted, unwrappedKey);
      expect(decrypted).toBe('test');
    });
  });

  describe('哈希函数', () => {
    it('应生成SHA-256哈希', async () => {
      const hash = await cryptoService.hash('password123');
      
      expect(hash).toBeDefined();
      expect(hash.length).toBe(64); // SHA-256 hex length
      expect(hash).toMatch(/^[a-f0-9]{64}$/);
    });

    it('相同输入应产生相同哈希', async () => {
      const hash1 = await cryptoService.hash('same input');
      const hash2 = await cryptoService.hash('same input');
      
      expect(hash1).toBe(hash2);
    });

    it('不同输入应产生不同哈希', async () => {
      const hash1 = await cryptoService.hash('input1');
      const hash2 = await cryptoService.hash('input2');
      
      expect(hash1).not.toBe(hash2);
    });
  });

  describe('错误处理', () => {
    it('应在无效密钥时抛出明确错误', async () => {
      const invalidKey = 'invalid-key-data';
      const encrypted = { iv: 'abc', data: 'def' };
      
      await expect(
        cryptoService.decrypt(encrypted, invalidKey as any)
      ).rejects.toThrow(/Invalid key/i);
    });

    it('应在数据损坏时抛出解密失败错误', async () => {
      const key = await cryptoService.generateKey();
      const corruptedData = { 
        iv: cryptoService.randomBytes(12), 
        data: 'corrupted-data-that-is-not-valid' 
      };
      
      await expect(
        cryptoService.decrypt(corruptedData as any, key)
      ).rejects.toThrow(/Decryption failed/i);
    });

    it('应在加密超大文本时处理内存限制', async () => {
      const key = await cryptoService.generateKey();
      // 1MB文本
      const largeText = 'x'.repeat(1024 * 1024);
      
      const startTime = performance.now();
      const encrypted = await cryptoService.encrypt(largeText, key);
      const duration = performance.now() - startTime;
      
      expect(encrypted).toBeDefined();
      expect(duration).toBeLessThan(1000); // 应该在1秒内完成
    }).timeout(5000);
  });

  describe('性能基准', () => {
    it('加密操作应在100ms内完成 (1KB数据)', async () => {
      const key = await cryptoService.generateKey();
      const data = 'x'.repeat(1024);
      
      const start = performance.now();
      for (let i = 0; i < 10; i++) {
        await cryptoService.encrypt(data, key);
      }
      const avgDuration = (performance.now() - start) / 10;
      
      expect(avgDuration).toBeLessThan(100);
    });

    it('密钥生成应在50ms内完成', async () => {
      const start = performance.now();
      await cryptoService.generateKey();
      const duration = performance.now() - start;
      
      expect(duration).toBeLessThan(50);
    });
  });
});

// Helper functions for creating mock streams
function createMockSSEStream(events: any[]) {
  return {
    async *[Symbol.asyncIterator]() {
      for (const event of events) {
        yield event;
      }
    },
    then(resolve: any) {
      resolve({ content: events.filter(e => e.type === 'token').map(e => e.content).join('') });
    }
  };
}

function createMockFailingStream(error: Error) {
  return {
    async *[Symbol.asyncIterator]() {
      throw error;
    },
    then() {
      return Promise.reject(error);
    }
  };
}

function createSlowSSEStream(events: any[], delayMs: number) {
  return {
    async *[Symbol.asyncIterator]() {
      for (const event of events) {
        await new Promise(r => setTimeout(r, delayMs));
        yield event;
      }
    }
  };
}
